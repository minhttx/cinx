-- Hàm SQL thực hiện xóa phim và toàn bộ dữ liệu liên quan (Cascade Delete)
-- Chạy lệnh này trong Supabase SQL Editor

CREATE OR REPLACE FUNCTION delete_movie_cascade(movie_uuid UUID)
RETURNS void AS $$
BEGIN
    -- 1. Xóa chi tiết ghế đặt (liên kết qua booking)
    DELETE FROM public.booking_seats 
    WHERE booking_id IN (
        SELECT id FROM public.bookings WHERE showtime_id IN (
            SELECT id FROM public.showtimes WHERE movie_id = movie_uuid
        )
    );

    -- 2. Xóa chi tiết ghế đặt (liên kết qua seat_id)
    DELETE FROM public.booking_seats 
    WHERE seat_id IN (
        SELECT id FROM public.seats WHERE showtime_id IN (
            SELECT id FROM public.showtimes WHERE movie_id = movie_uuid
        )
    );

    -- 3. Xóa đơn hàng
    DELETE FROM public.bookings WHERE showtime_id IN (
        SELECT id FROM public.showtimes WHERE movie_id = movie_uuid
    );
    
    -- Xóa đơn hàng nếu có liên kết trực tiếp movie_id
    DELETE FROM public.bookings WHERE movie_id::text = movie_uuid::text;

    -- 4. Xóa ghế
    DELETE FROM public.seats WHERE showtime_id IN (
        SELECT id FROM public.showtimes WHERE movie_id = movie_uuid
    );

    -- 5. Xóa suất chiếu
    DELETE FROM public.showtimes WHERE movie_id = movie_uuid;

    -- 6. Xóa phim
    DELETE FROM public.movies WHERE id = movie_uuid;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
